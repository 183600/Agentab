连接信息

其他扩展可以通过以下方式连接 Agentab MCP Server:

const extensionId = 'YOUR_AGENTAB_EXTENSION_ID';

// 发送 MCP 请求
chrome.runtime.sendMessage(
  extensionId,
  {
    jsonrpc: '2.0',
    id: 1,
    method: 'tools/list',
    params: {}
  },
  (response) => {
    console.log('Available tools:', response.result.tools);
  }
);

// 调用工具
chrome.runtime.sendMessage(
  extensionId,
  {
    jsonrpc: '2.0',
    id: 2,
    method: 'tools/call',
    params: {
      name: 'save_task',
      arguments: {
        name: 'My Task',
        type: 'prompt',
        content: 'Hello world'
      }
    }
  },
  (response) => {
    console.log('Result:', response.result);
  }
);
